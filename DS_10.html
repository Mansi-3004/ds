outliers=[]
for x in sort_data:
    z = (x - mean)/std
    if z >2:
        outliers.append(x)

print("Q1 : ",Q1,"\n","Q3: ",Q3)
 print("inter quartile range : ", IQR)
 low_limit = Q1-1.5*IQR
 up_limit = Q3+1.5*IQR
 print('low_limit is', low_limit)
 print('up_limit is', up_limit)
 outliers = []
 for x in sort_data2:
    if ((x < low_limit) or (x > up_limit)):
